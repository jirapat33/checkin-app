// ===== รับ POST จาก GitHub Pages ===== function doPost(e) { try { var data = JSON.parse(e.postData.contents); var result = processCheckIn(data); return ContentService .createTextOutput(result) .setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService .createTextOutput("ERROR: " + err.toString()) .setMimeType(ContentService.MimeType.TEXT); } } // ===== doGet — แสดงข้อความแทน (หน้าจริงอยู่ที่ GitHub Pages แล้ว) ===== function doGet() { return ContentService.createTextOutput("Apps Script Backend พร้อมใช้งาน ✅"); } // ===== ฟังก์ชันบันทึกข้อมูลลง Google Sheet ===== function processCheckIn(data) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("ปวส2/1"); if (!sheet) return "ERROR: ไม่พบแผ่นงาน 'ปวส2/1'"; var lastRow = sheet.getLastRow(); if (lastRow < 2) return "ERROR: ไม่มีข้อมูลรายชื่อนักศึกษา"; var studentIdRange = sheet.getRange(2, 3, lastRow - 1, 1).getValues(); var targetRow = -1; var inputId = data.studentId.toString().trim(); for (var i = 0; i < studentIdRange.length; i++) { if (studentIdRange[i][0].toString().trim() === inputId) { targetRow = i + 2; break; } } if (targetRow === -1) return "ERROR: ไม่พบรหัสนักศึกษา [" + inputId + "]"; // อัปโหลดรูปภาพ var folder = DriveApp.getRootFolder(); var contentType = data.image.split(',')[0].split(':')[1].split(';')[0]; var bytes = Utilities.base64Decode(data.image.split(',')[1]); var blob = Utilities.newBlob(bytes, contentType, "CheckIn_" + inputId + "_" + Date.now() + ".jpg"); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); // บันทึกข้อมูล sheet.getRange(targetRow, 1).setValue(new Date()); sheet.getRange(targetRow, 5).setValue(data.lat); sheet.getRange(targetRow, 6).setValue(data.lon); sheet.getRange(targetRow, 7).setValue(file.getUrl()); return "SUCCESS"; } catch (err) { return "ERROR: " + err.toString(); } }
